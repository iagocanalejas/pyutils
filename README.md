````bash
rg --glob "*.py" -N "^def " pyutils | sort | \
 awk -F: '
{
file=$1
line=$0
sub(/^[^:]+:/, "", line)
sub(/^ *def /, "", line)
if (file != last_file) {
if (last_file != "") print "`"
      print "\n" file "\n`python"
last_file=file
}
print line
}
END { print "```" }' >> README.md
````
